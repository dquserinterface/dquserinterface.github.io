47def listToString(s):
48 """
49 ListToString provides converts lists to strings with commas.
50 This function is written to save as string type instead of list
51
52
53 Args:
54 s (list): Input as List
55
56 Returns:
57 string: Comma seperated string
58 """
59 if len(s) > 1:
60 # initialize an empty string
61 str1 = ","
62
63 # return string
64 return str1.join(s)
65 else:
66 str1 = " "
67
68 return str1.join(s)

104class ChoicesAction(argparse._StoreAction):
105 """
106 ChoicesAction class is used to add extra choices
107 to a parseargs choices list
108
109 Args:
110 argparse (Class): Input as args
111 """
112
113 def add_choice(self, choice, help=""):
114 if self.choices is None:
115 self.choices = []
116 self.choices.append(choice)
117 self.container.add_argument(choice, help=help, action="none")
